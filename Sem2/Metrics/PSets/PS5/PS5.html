X <- as.matrix(cbind(1, male_dat%>%select(age, education, black, hisp)))
colnames(X) <- c('intercept', 'age', 'educ', 'black', 'hisp')
Y <- male_dat$Y

probit <- function(beta, X, Y) {
  xbeta <- X %*% beta
  prob <- pnorm(xbeta)
  prob <- pmax(pmin(prob, 1-1e-10), 1e-10)
  sum(Y*log(prob) + (1-Y)*log(1-prob))
}

neg_probit <- function(beta, X, Y) {
  -probit(beta, X, Y)
}

beta_init <- rep(0, ncol(X))
result <- optim(beta_init, neg_probit, X = X, Y = Y, method = "BFGS",
                control = list(maxit = 1000), hessian = TRUE)

beta_hat <- result$par

H <- result$hessian
H_reg <- H + diag(1e-8, ncol(X))
vcov_beta <- solve(H_reg)
se_beta <- sqrt(diag(vcov_beta))

coef_table <- data.frame(
  Variable = colnames(X),
  Coefficient = beta_hat,
  Std_Error = se_beta,
  z_value = beta_hat/se_beta,
  p_value = 2*pnorm(-abs(beta_hat/se_beta))
)
